def prim(graph, V, labels):
    selected = [False] * V
    selected[0] = True
    edges = 0
    total_cost = 0

    print("\nEdge : Weight")

    while edges < V - 1:
        minimum = float('inf')
        x = 0
        y = 0

        for i in range(V):
            if selected[i]:
                for j in range(V):
                    if not selected[j] and graph[i][j] != 0:
                        if graph[i][j] < minimum:
                            minimum = graph[i][j]
                            x = i
                            y = j

        print(f"{labels[x]} - {labels[y]} : {graph[x][y]}")
        total_cost += graph[x][y]
        selected[y] = True
        edges += 1

    print("Total Weight of MST =", total_cost)


V = int(input("Enter number of vertices: "))

labels = input("Enter vertex labels separated by space: ").split()

print("Enter adjacency matrix row by row:")

graph = []

for i in range(V):
    row = list(map(int, input().split()))
    graph.append(row)

prim(graph, V, labels)
